Next we iterated through each chromosome, and identified the number of differentially expressed transcripts on each using a quasipoisson generalized linear model

for (chr in chrs) {
  DEs = vector() ## This holds the DE transcripts for the chromosome
  
  ### first get relevent TPM matrix ###
  TPM_mat <- paste("expr_chr", chr, sep = "")
  TPM_mat <- get(TPM_mat)
  
  for (row in 1:nrow(TPM_mat)) {
    tran <- TPM_mat[row,] # grab each set of expression data per transcript ###
    tran <- (as.data.frame(t(as.matrix(tran)))) # transpose the data
    colnames(tran)[1] <- "TPM" 
    tran$cell_type <- c(1,1,0,0) # add a column for cell type (1=brain, 0=stem)
    
    ### We ignore all cases where expression was all zeros ###
    check <- list()
    for (i in 1:nrow(tran)) {
      if (tran[i, 1] != 0) {
        check <- append(check, 1)
      }
    }
    
    ### Make a quasipoisson logistic model to test if expression level predicts cell type
    if (length(check) > 0) {
      model <- glm(TPM ~ cell_type, data = tran, family = quasipoisson)
    }
    
    ### If p-value less than .05, we conclude cell type predicts expression ### 
    pVal <- (as.matrix(summary(model)$coefficients)[2,4])
    if (as.numeric(pVal) < .05){
      DEs <- append(DEs, rownames(TPM_mat[row,]))
    }
  }
  
  ### save our raw counts for the chromosome ###
  qp_DEs <- append(qp_DEs,length(DEs)) 
  ### save our standardized results by dividing by chromosome length ###
  standDE <- append(standDE, length(DEs)/nrow(ESCd1[(ESCd1$chromosome==chr),]))
}

The resulting p value of the chi square is equal to 1. This means that we fail to reject our null hypothesis; strongly indicating that the differential expression counts are equal across all chromosomes and any differences are due to chance alone.
